"use client";

import { useEffect, useState } from "react";
import Link from "next/link";

export default function AddToCart() {
  const [cart, setCart] = useState<any[]>([]);
  const [loading, setLoading] = useState(true);

  // ‚úÖ 1. T·∫°o guest_id n·∫øu ch∆∞a c√≥ (ch·ªâ ch·∫°y 1 l·∫ßn)
  useEffect(() => {
    const guestId = localStorage.getItem("guest_id");
    if (!guestId) {
      fetch("http://localhost:3000/api/auth/guest", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
      })
        .then((res) => res.json())
        .then((data) => {
          if (data.success && data.guest?.id) {
            localStorage.setItem("guest_id", data.guest.id);
            console.log("‚úÖ T·∫°o guest_id th√†nh c√¥ng:", data.guest.id);
          }
        })
        .catch((err) => console.error("‚ùå L·ªói t·∫°o user kh√°ch:", err));
    }
  }, []);

  // ‚úÖ 2. H√†m th√™m gi·ªè h√†ng
  const handleAddToCart = async (productId: number, quantity: number) => {
    const user = JSON.parse(localStorage.getItem("user") || "null");
    const guestId = localStorage.getItem("guest_id");
    const id_user = user ? user.id : guestId;

    if (!id_user) {
      alert("Kh√¥ng th·ªÉ th√™m v√†o gi·ªè h√†ng (ch∆∞a c√≥ ID ng∆∞·ªùi d√πng)!");
      return;
    }

    const res = await fetch("http://localhost:3000/api/cart/add", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        id_user,
        id_san_pham: productId,
        so_luong: quantity,
      }),
    });

    const data = await res.json();
    console.log("üõçÔ∏è Th√™m gi·ªè h√†ng:", data);
    fetchCart(); // load l·∫°i gi·ªè
  };

  // ‚úÖ 3. L·∫•y gi·ªè h√†ng (guest ho·∫∑c user)
  const fetchCart = async () => {
    setLoading(true);
    try {
      const user = JSON.parse(localStorage.getItem("user") || "null");
      const guestId = localStorage.getItem("guest_id");
      const id_user = user ? user.id : guestId;

      if (!id_user) {
        setCart([]);
        return;
      }

      const res = await fetch(`http://localhost:3000/api/cart?id_user=${id_user}`, {
        credentials: "include",
      });

      if (!res.ok) throw new Error("L·ªói fetch gi·ªè h√†ng");
      const data = await res.json();

      setCart(Array.isArray(data) ? data : []);
    } catch (err) {
      console.error("üö® L·ªói t·∫£i gi·ªè h√†ng:", err);
      setCart([]);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchCart();
  }, []);

  // ‚úÖ X√≥a s·∫£n ph·∫©m
  const handleDelete = async (id: number) => {
    try {
      const res = await fetch(`http://localhost:3000/api/cart/delete/${id}`, {
        method: "DELETE",
      });

      if (res.ok) {
        setCart((prev) => prev.filter((item) => (item.id || item.id_san_pham) !== id));
      }
    } catch (err) {
      console.error("‚ùå L·ªói x√≥a s·∫£n ph·∫©m:", err);
    }
  };

  // ‚úÖ C·∫≠p nh·∫≠t s·ªë l∆∞·ª£ng
  const handleQuantityChange = async (id: number, newQty: number) => {
    if (newQty < 1) return;

    try {
      const res = await fetch(`http://localhost:3000/api/cart/update/${id}`, {
        method: "PUT",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ so_luong: newQty }),
      });

      if (res.ok) {
        setCart((prev) =>
          prev.map((item) =>
            (item.id || item.id_san_pham) === id ? { ...item, so_luong: newQty } : item
          )
        );
      }
    } catch (err) {
      console.error("‚ùå L·ªói c·∫≠p nh·∫≠t s·ªë l∆∞·ª£ng:", err);
    }
  };

  // ‚úÖ ƒê·∫∑t h√†ng
  const handleOrder = async () => {
    const user = JSON.parse(localStorage.getItem("user") || "null");
    const guestId = localStorage.getItem("guest_id");
    const id_user = user ? user.id : guestId;

    const res = await fetch("http://localhost:3000/api/order", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ id_user }),
    });

    const data = await res.json();
    console.log("üßæ K·∫øt qu·∫£ ƒë·∫∑t h√†ng:", data);
  };

  // üõí Giao di·ªán hi·ªÉn th·ªã
  if (loading) return <p>ƒêang t·∫£i gi·ªè h√†ng...</p>;
  if (cart.length === 0) return <h1>üõí Gi·ªè h√†ng tr·ªëng</h1>;

  const tongTien = cart.reduce(
    (sum, item) => sum + (Number(item.gia) || 0) * (Number(item.so_luong) || 0),
    0
  );

  return (
    <div className="max-w-5xl mx-auto my-10 bg-white p-6 rounded-xl shadow-lg">
      <h2 className="text-3xl font-bold mb-6 text-center text-blue-700">
        üõçÔ∏è Gi·ªè H√†ng C·ªßa B·∫°n
      </h2>

      {/* Header b·∫£ng */}
      <div className="hidden md:grid grid-cols-[80px_1fr_120px_120px_80px] gap-4 font-semibold border-b pb-2 mb-4">
        <div>·∫¢nh</div>
        <div>T√™n s·∫£n ph·∫©m</div>
        <div>Gi√°</div>
        <div>S·ªë l∆∞·ª£ng</div>
        <div></div>
      </div>

      {/* Danh s√°ch s·∫£n ph·∫©m */}
      {cart.map((item: any) => (
        <div
          key={item.id || item.id_san_pham}
          className="grid grid-cols-[80px_1fr_120px_120px_80px] items-center gap-4 p-3 border-b hover:bg-gray-50 transition-colors duration-200 rounded-lg md:rounded-none"
        >
          <img
            src={item.hinh}
            alt={item.ten_san_pham}
            className="w-21 h-21 object-cover rounded-lg"
          />
          <div>
            <h3 className="font-semibold">{item.ten_san_pham}</h3>
            <p className="text-gray-500 text-sm">M√†u: {item.mau_sac}</p>
          </div>
          <div className="text-blue-600 font-bold">
            {Number(item.gia).toLocaleString()} ‚Ç´
          </div>
          <div className="flex items-center">
            <button
              className="px-2 py-1 bg-gray-200 rounded-l hover:bg-gray-300 transition-colors"
              onClick={() =>
                handleQuantityChange(item.id || item.id_san_pham, item.so_luong - 1)
              }
            >
              -
            </button>
            <span className="px-3">{item.so_luong}</span>
            <button
              className="px-2 py-1 bg-gray-200 rounded-r hover:bg-gray-300 transition-colors"
              onClick={() =>
                handleQuantityChange(item.id || item.id_san_pham, item.so_luong + 1)
              }
            >
              +
            </button>
          </div>
          <button
            className="text-red-600 hover:underline font-semibold transition-colors"
            onClick={() => handleDelete(item.id || item.id_san_pham)}
          >
            X√≥a
          </button>
        </div>
      ))}

      {/* T·ªïng ti·ªÅn */}
      <div className="text-right mt-6 text-xl font-bold text-black-700">
        T·ªïng ti·ªÅn: {tongTien.toLocaleString("vi-VN")} ‚Ç´
      </div>

      {/* N√∫t thanh to√°n */}
      <div className="text-right mt-4">
        <Link href="/Checkout">
          <button className="bg-blue-600 text-white px-6 py-2 rounded-lg font-semibold hover:bg-blue-700 transition-colors shadow-md hover:shadow-lg">
            ƒê·∫∑t H√†ng
          </button>
        </Link>
      </div>
    </div>
  );
}
